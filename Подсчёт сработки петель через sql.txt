select * from ev_bas_logg 
where WDEVICEID = 202 
and CEVENTTRANSPTEXT like '%Петли: E:o  Z:o  S:x%' 
and teventtimest between '01-NOV-20 17:00:00' and '02-NOV-20 17:00:00'
order by TEVENTTIMEST;


select wdeviceid as устройство, count(wdeviceid) as кол_сработок from ev_bas_logg 
where CEVENTTRANSPTEXT like '%Петли: E:o  Z:o  S:x%'
and teventtimest between '01-NOV-20 17:00:00' and '04-NOV-20 17:00:00'
group by wdeviceid
order by wdeviceid;

select wdeviceid as устройство, count(wdeviceid) as кол_сработок from ev_bas_logg 
where WDEVICEID between 339 and 352
and CEVENTTRANSPTEXT like '%Петли: E:o  Z:o  S:x%'
and teventtimest between '01-NOV-20 17:00:00' and '04-NOV-20 17:00:00'
group by wdeviceid
order by wdeviceid;
